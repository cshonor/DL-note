# 防过拟合三件套：最小可运行示例（NumPy）

> 目标：用一个极简可跑例子，把 `L2 正则 + Dropout + Early Stopping` 一次串起来。  
> 网络结构按你要求：`输入 -> 5 -> 4 -> 3 -> 2 -> 1`（线性层）。

---

## 一、先看完整代码

```python
import numpy as np

# --------------- 1. 简单造一点假数据 -------------------
# 随便造一组数据：100个样本，每个样本4个特征
X = np.random.randn(100, 4)
y = np.random.randn(100, 1)

# 划分：训练集 + 验证集（Early Stopping 必须要有验证集）
X_train, y_train = X[:80], y[:80]
X_val, y_val = X[80:], y[80:]


# --------------- 2. 激活函数这里用线性 -------------------
def linear(x):
    return x


def linear_deriv(x):
    return np.ones_like(x)


# --------------- 3. 初始化网络结构：4->5->4->3->2->1 -------------------
input_dim = 4
hidden1 = 5
hidden2 = 4
hidden3 = 3
hidden4 = 2
output_dim = 1

np.random.seed(42)
W1 = np.random.randn(input_dim, hidden1) * 0.01
b1 = np.zeros((1, hidden1))

W2 = np.random.randn(hidden1, hidden2) * 0.01
b2 = np.zeros((1, hidden2))

W3 = np.random.randn(hidden2, hidden3) * 0.01
b3 = np.zeros((1, hidden3))

W4 = np.random.randn(hidden3, hidden4) * 0.01
b4 = np.zeros((1, hidden4))

W5 = np.random.randn(hidden4, output_dim) * 0.01
b5 = np.zeros((1, output_dim))


# --------------- 4. Dropout 函数 -------------------
def dropout(x, dropout_rate=0.2):
    mask = np.random.rand(*x.shape) > dropout_rate  # 随机关掉一些神经元
    x = x * mask / (1 - dropout_rate)  # 训练时缩放，保证输出均值近似不变
    return x, mask


# --------------- 5. 超参数（正则 + 早停）-------------------
lr = 0.01
l2_lambda = 0.001
dropout_rate = 0.2
epochs = 10000

# Early Stopping 配置
best_val_loss = float("inf")
patience = 20
patience_counter = 0
best_weights = None


# --------------- 6. 训练主循环 -------------------
for epoch in range(epochs):
    # -------- 前向传播（训练：带 Dropout）--------
    z1 = X_train @ W1 + b1
    a1 = linear(z1)
    a1, mask1 = dropout(a1, dropout_rate)

    z2 = a1 @ W2 + b2
    a2 = linear(z2)
    a2, mask2 = dropout(a2, dropout_rate)

    z3 = a2 @ W3 + b3
    a3 = linear(z3)
    a3, mask3 = dropout(a3, dropout_rate)

    z4 = a3 @ W4 + b4
    a4 = linear(z4)

    z5 = a4 @ W5 + b5
    y_hat = linear(z5)

    # -------- 训练损失：MSE + L2 --------
    mse_loss = np.mean((y_hat - y_train) ** 2)
    l2_loss = 0.5 * l2_lambda * (
        np.sum(W1**2)
        + np.sum(W2**2)
        + np.sum(W3**2)
        + np.sum(W4**2)
        + np.sum(W5**2)
    )
    total_loss = mse_loss + l2_loss

    # -------- 验证前向（验证：不使用 Dropout）--------
    val_z1 = X_val @ W1 + b1
    val_a1 = linear(val_z1)

    val_z2 = val_a1 @ W2 + b2
    val_a2 = linear(val_z2)

    val_z3 = val_a2 @ W3 + b3
    val_a3 = linear(val_z3)

    val_z4 = val_a3 @ W4 + b4
    val_a4 = linear(val_z4)

    val_z5 = val_a4 @ W5 + b5
    val_y_hat = linear(val_z5)
    val_loss = np.mean((val_y_hat - y_val) ** 2)

    # -------- Early Stopping --------
    if val_loss < best_val_loss:
        best_val_loss = val_loss
        patience_counter = 0
        best_weights = (
            W1.copy(),
            b1.copy(),
            W2.copy(),
            b2.copy(),
            W3.copy(),
            b3.copy(),
            W4.copy(),
            b4.copy(),
            W5.copy(),
            b5.copy(),
        )
    else:
        patience_counter += 1

    if patience_counter >= patience:
        print(f"Early Stopping at epoch {epoch}")
        if best_weights is not None:
            W1, b1, W2, b2, W3, b3, W4, b4, W5, b5 = best_weights
        break

    # -------- 反向传播（含 L2 梯度）--------
    delta5 = (y_hat - y_train) * linear_deriv(z5)
    delta4 = delta5 @ W5.T * linear_deriv(z4)
    delta3 = delta4 @ W4.T * linear_deriv(z3) * mask3 / (1 - dropout_rate)
    delta2 = delta3 @ W3.T * linear_deriv(z2) * mask2 / (1 - dropout_rate)
    delta1 = delta2 @ W2.T * linear_deriv(z1) * mask1 / (1 - dropout_rate)

    # 权重更新 + L2
    W5 -= lr * (a4.T @ delta5 + l2_lambda * W5)
    b5 -= lr * np.sum(delta5, axis=0, keepdims=True)

    W4 -= lr * (a3.T @ delta4 + l2_lambda * W4)
    b4 -= lr * np.sum(delta4, axis=0, keepdims=True)

    W3 -= lr * (a2.T @ delta3 + l2_lambda * W3)
    b3 -= lr * np.sum(delta3, axis=0, keepdims=True)

    W2 -= lr * (a1.T @ delta2 + l2_lambda * W2)
    b2 -= lr * np.sum(delta2, axis=0, keepdims=True)

    W1 -= lr * (X_train.T @ delta1 + l2_lambda * W1)
    b1 -= lr * np.sum(delta1, axis=0, keepdims=True)

    if epoch % 100 == 0:
        print(
            f"Epoch {epoch:4d} | train_mse: {mse_loss:.4f} | "
            f"train_total: {total_loss:.4f} | val_loss: {val_loss:.4f}"
        )

print("Training finished. Using best validation checkpoint.")
```

---

## 二、这段代码对应第8章哪三个点

### 1) L2 正则
- 损失里加了 `0.5 * lambda * sum(W^2)`。
- 更新时每层都加了 `+ l2_lambda * W` 梯度项。
- 作用是抑制权重过大，降低过拟合风险。

### 2) Dropout
- 训练前向在前几层做随机失活。
- 验证阶段完全关闭 Dropout（必须）。
- 作用是打散神经元共适应，减少抱团依赖。

### 3) Early Stopping
- 每轮看验证集 `val_loss`。
- 若连续 `patience` 轮没提升就停训。
- 自动回退到验证最优权重。

---

## 三、一句口诀

**L2 管权重大小，Dropout 管特征依赖，Early Stopping 管训练别过头。**
